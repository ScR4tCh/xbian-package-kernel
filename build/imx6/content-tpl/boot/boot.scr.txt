setenv fstype btrfs

setenv baseconfig console=tty0 console=ttymxc0,115200 coherent_pool=1M rw ahci_imx.hotplug=1 raid=noautodetect telnet root=/dev/mmcblk0p2 video=mxcfb0:dev=hdmi,1920x1080M@60,if=RGB24 mxc_hdmi.rgb_quant_range=full rootwait consoleblank=0 selinux=0 loglevel=7 nohdparm splash partswap dmfc=3

setenv cmasize cma=256M
if test ${cpu} = 6SOLO && test ${board} != mx6-cubox-i; then
    setenv cmasize cma=256M
fi

if test ${fstype} != btrfs; then
    setenv bootargs $baseconfig $cmasize rootfstype=ext4 rootflags=data=writeback --startup-event mountall
else
    setenv bootargs $baseconfig $cmasize rootfstype=btrfs rootflags=subvol=root/@,autodefrag,compress=lz4 --startup-event mountall
fi

if test ${debug} != ""; then exit 1; fi

load mmc 0:1 ${loadaddr} zImage

run autodetectfdt
load mmc 0:1 ${fdt_addr} ${fdt_file}

if load mmc 0:1 0x19000000 initramfs.gz; then
    setenv initram 0x19000000
else
    setenv initram "-"
fi

if test ${debug} != ""; then exit 1; fi

bootz ${loadaddr} ${initram} ${fdt_addr}
